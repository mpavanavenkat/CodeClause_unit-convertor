const contentDiv = card.querySelector('.testimonial-content'); const nameEl = contentDiv.querySelector('.name-display'); const messageEl = contentDiv.querySelector('.message-display'); const editBtn = card.querySelector('.edit-btn'); card.classList.add('editing'); } } // Add listeners to the pre-existing testimonial's buttons document.querySelectorAll('.edit-btn').forEach(btn => btn.addEventListener('click', handleEdit)); document.querySelectorAll('.delete-btn').forEach(btn => btn.addEventListener('click', handleDelete));
